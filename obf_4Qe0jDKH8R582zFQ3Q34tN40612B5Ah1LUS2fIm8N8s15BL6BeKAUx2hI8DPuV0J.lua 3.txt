local function safeLoadstring(url)
    local success, result = pcall(function()
        if syn and syn.request then
            local response = syn.request({Url = url, Method = "GET"})
            if response and response.Body then
                return loadstring(response.Body)
            end
        elseif request then
            local response = request({Url = url, Method = "GET"})
            if response and response.Body then
                return loadstring(response.Body)
            end
        else
            local response = game:HttpGet(url)
            if response then
                return loadstring(response)
            end
        end
    end)
    if success and result then
        return result
    else
        warn("Failed to load: " .. url)
        return function() end
    end
end

-- Safe HTTP function for all executors
local function httpGet(url)
    local success, result = pcall(function()
        if syn and syn.request then
            local response = syn.request({Url = url, Method = "GET"})
            return response.Body
        elseif request then
            local response = request({Url = url, Method = "GET"})
            return response.Body
        else
            return game:HttpGet(url)
        end
    end)
    return success and result or nil
end


-- Safe setclipboard for all executors
local function safeSetClipboard(text)
    pcall(function()
        if setclipboard then
            setclipboard(text)
        elseif toclipboard then
            toclipboard(text)
        elseif writeclipboard then
            writeclipboard(text)
        end
    end)
end

-- ESP Functionality
local function createESP(plr)
    if plr == player then return end
    if plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
        if espBoxes[plr] then
            if espBoxes[plr].Box then pcall(function() espBoxes[plr].Box:Destroy() end) end
            if espBoxes[plr].Text then pcall(function() espBoxes[plr].Text:Destroy() end) end
        end
        local box = Instance.new("BoxHandleAdornment")
        box.Adornee = plr.Character.HumanoidRootPart
        box.AlwaysOnTop = true
        box.ZIndex = 10
        box.Size = Vector3.new(2, plr.Character.HumanoidRootPart.Size.Y + 3, 2)
        box.Color3 = Color3.fromRGB(255,255,255)
        box.Transparency = 0.7
        box.Parent = plr.Character.HumanoidRootPart
        local billboard = Instance.new("BillboardGui")
        billboard.Size = UDim2.new(0,250,0,50)
        billboard.AlwaysOnTop = true
        billboard.Adornee = plr.Character.HumanoidRootPart
        billboard.Parent = player:WaitForChild("PlayerGui")
        local text = Instance.new("TextLabel")
        text.Size = UDim2.new(1,0,1,0)
        text.BackgroundTransparency = 1
        text.TextColor3 = Color3.fromRGB(255,255,255)
        text.TextScaled = true
        text.Font = Enum.Font.GothamBold
        text.TextWrapped = true
        text.Parent = billboard
        espBoxes[plr] = {Box=box, Text=text}
    end
end


-- Run Service for ESP and Movement
game:GetService("RunService").RenderStepped:Connect(function()
    if espEnabled then
        for _, plr in pairs(game.Players:GetPlayers()) do
            if plr ~= player and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
                createESP(plr)
                local hrp = plr.Character.HumanoidRootPart
                local dist = (player.Character.HumanoidRootPart.Position - hrp.Position).Magnitude
                if espBoxes[plr] and espBoxes[plr].Text then
                    espBoxes[plr].Text.Text = string.format("%s | %s | %.1f studs", plr.Name, plr.DisplayName, dist)
                end
            end
        end
    end
    if player.Character and player.Character:FindFirstChild("Humanoid") then
        local hum = player.Character.Humanoid
        if hum.WalkSpeed ~= targetWalkSpeed then hum.WalkSpeed = targetWalkSpeed end
        if hum.JumpPower ~= targetJumpPower then hum.JumpPower = targetJumpPower end
    end
end)


player.CharacterAdded:Connect(function(char)
    local hum = char:WaitForChild("Humanoid")
    hum.WalkSpeed = targetWalkSpeed
    hum.JumpPower = targetJumpPower
end)


game.Players.PlayerAdded:Connect(function(plr)
    plr.CharacterAdded:Connect(function()
        if espEnabled then task.wait(0.5) createESP(plr) end
    end)
end)
